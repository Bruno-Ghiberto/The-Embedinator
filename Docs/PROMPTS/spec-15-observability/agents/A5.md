### T043 — `frontend/hooks/useMetrics.ts` (NEW)

SWR hook wrapping `GET /api/metrics`:
```typescript
export function useMetrics(window: "1h" | "24h" | "7d" = "24h") {
  const { data, error, isLoading } = useSWR<MetricsResponse>(
    `/api/metrics?window=${window}`,
    fetcher
  )
  return { data, error, isLoading }
}
```

Define the `MetricsResponse` TypeScript type (matching the backend Pydantic model):
```typescript
interface MetricsBucket {
  timestamp: string
  query_count: number
  avg_latency_ms: number
  p95_latency_ms: number
  avg_confidence: number
  meta_reasoning_count: number
  error_count: number
}

interface CircuitBreakerSnapshot {
  state: "closed" | "open" | "unknown"
  failure_count: number
}

interface MetricsResponse {
  window: string
  bucket_size: string
  buckets: MetricsBucket[]
  circuit_breakers: Record<string, CircuitBreakerSnapshot>
  active_ingestion_jobs: number
}
```
